```html
<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>فانيليا قريش</title>
    <script src="https://cdn.tailwindcss.com"></script>
    <style>
        body { font-family: 'Tajawal', sans-serif; background-color: #fcfcfc; }
    </style>
</head>
<body class="p-4">

    <div class="max-w-md mx-auto">
        <header class="text-center py-6">
            <h1 class="text-2xl font-black text-[#4a2c1d]">فانيليا قريش</h1>
        </header>

        <div class="grid grid-cols-2 gap-4 mb-6">
            <div class="bg-stone-100 p-4 rounded-2xl text-center">
                <span class="text-xs text-stone-500">المدفوع</span>
                <div id="sum-paid" class="text-xl font-bold">0</div>
            </div>
            <div class="bg-[#4a2c1d] p-4 rounded-2xl text-center text-white">
                <span class="text-xs opacity-70">المتبقي</span>
                <div id="sum-rem" class="text-xl font-bold">0</div>
            </div>
        </div>

        <button onclick="openModal()" class="w-full bg-[#4a2c1d] text-white p-4 rounded-2xl font-bold mb-6">+ إضافة صفقة</button>
        <div id="list" class="space-y-4"></div>
    </div>

    <div id="modal" class="fixed inset-0 bg-black/50 hidden p-4 flex items-center justify-center">
        <div class="bg-white p-6 rounded-3xl w-full max-w-sm">
            <input type="hidden" id="edit_id">
            <input type="number" id="tx_qty" placeholder="الكمية" class="w-full p-3 bg-stone-100 rounded-xl mb-3">
            <input type="number" id="tx_paid" placeholder="المقدم لكل كيلو" class="w-full p-3 bg-stone-100 rounded-xl mb-3">
            <select id="tx_ship" onchange="toggleTax()" class="w-full p-3 bg-stone-100 rounded-xl mb-3">
                <option value="مسافرين">شحن: مسافرين</option>
                <option value="شحن جوي">شحن: جوي</option>
            </select>
            <div id="tax_section" class="hidden">
                <input type="number" id="tx_tax" placeholder="الضريبة" class="w-full p-3 bg-stone-100 rounded-xl mb-3">
                <select id="tx_payer" class="w-full p-3 bg-stone-100 rounded-xl mb-3">
                    <option value="قريش">الضريبة على قريش</option>
                    <option value="أنا">الضريبة عليّ أنا</option>
                </select>
            </div>
            <button onclick="save()" class="w-full bg-[#4a2c1d] text-white p-3 rounded-xl font-bold">حفظ</button>
        </div>
    </div>

    <script>
        let txs = JSON.parse(localStorage.getItem('txs') || '[]');

        function toggleTax() {
            document.getElementById('tax_section').classList.toggle('hidden', document.getElementById('tx_ship').value === 'مسافرين');
        }

        function openModal(id = null) {
            document.getElementById('modal').classList.remove('hidden');
            if (id) {
                let t = txs.find(x => x.id === id);
                document.getElementById('edit_id').value = id;
                document.getElementById('tx_qty').value = t.q;
                document.getElementById('tx_paid').value = t.paid;
                document.getElementById('tx_ship').value = t.ship;
                toggleTax();
                if(t.ship === 'شحن جوي') {
                    document.getElementById('tx_tax').value = t.tax;
                    document.getElementById('tx_payer').value = t.payer;
                }
            } else {
                document.getElementById('edit_id').value = '';
            }
        }

        function save() {
            let id = document.getElementById('edit_id').value;
            let q = +document.getElementById('tx_qty').value;
            let paid = +document.getElementById('tx_paid').value;
            let ship = document.getElementById('tx_ship').value;
            let tax = (ship === 'شحن جوي') ? +document.getElementById('tx_tax').value : 0;
            let payer = (ship === 'شحن جوي') ? document.getElementById('tx_payer').value : 'قريش';
            
            let rem = (450 - paid) * q;
            if (ship === 'شحن جوي' && payer === 'أنا') rem -= tax;

            if (id) {
                let index = txs.findIndex(x => x.id == id);
                txs[index] = { id: +id, q, paid, ship, tax, payer, rem };
            } else {
                txs.push({ id: Date.now(), q, paid, ship, tax, payer, rem });
            }
            localStorage.setItem('txs', JSON.stringify(txs));
            location.reload();
        }

        function render() {
            let sp = 0, sr = 0;
            txs.forEach(t => {
                sp += (t.paid * t.q);
                sr += t.rem;
                document.getElementById('list').innerHTML += `
                    <div onclick="openModal(${t.id})" class="p-4 bg-white rounded-2xl shadow-sm border flex justify-between">
                        <div>الكمية: ${t.q} كجم</div>
                        <div class="font-bold text-[#4a2c1d]">المتبقي: ${t.rem}</div>
                    </div>`;
            });
            document.getElementById('sum-paid').innerText = sp;
            document.getElementById('sum-rem').innerText = sr;
        }
        render();
    </script>
</body>
</html>

```
